        function loadClients() {
            const clients = JSON.parse(localStorage.getItem('eliteFamilyClients')) || [];
            const clientList = document.getElementById('clientList');
            const clientCount = document.getElementById('clientCount');
            
            clientCount.textContent = `(${clients.length})`;
            
            if (clients.length === 0) {
                clientList.innerHTML = '<div class="client-item" style="text-align: center; color: #666;"><p>No clients yet. Click "Add New Client" to get started.</p></div>';
                return;
            }

                const clientItem = document.createElement('div');

            });
        }
